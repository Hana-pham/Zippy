function submitForm(event) {
    event.preventDefault(); // stop the from submit
    var url = document.getElementById('inputUrl').value; // get the inpuut Url

    //  sent the POST to the python code to check the url is work or not
    fetch('/check-url', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/x-www-form-urlencoded'
        },
        body: 'url=' + encodeURIComponent(url) //rewrite the Url
    })
    .then(response => response.json()) // change the response to a json model
    .then(data => {
        // refresh page
        document.getElementById('result').innerText = data.message;
        // if the web said can go in, the use differen css
        if (data.status === 'success') {
            document.getElementById('inputUrl').classList.remove('shake','border-red');
        } else {
            document.getElementById('inputUrl').classList.add('shake','border-red');
            setTimeout(() => {
        document.getElementById('inputUrl').classList.remove('border-red');
      }, 2000); // after 2s remove the red

        }
    })
    .catch(error => {
        // if cannot access, show error
        document.getElementById('result').innerText = 'Error: Failed to check the URL.';
        console.error('Error:', error);
        return;
    });
}

// sumit the form
document.getElementById('urlForm').addEventListener('submit', submitForm);
